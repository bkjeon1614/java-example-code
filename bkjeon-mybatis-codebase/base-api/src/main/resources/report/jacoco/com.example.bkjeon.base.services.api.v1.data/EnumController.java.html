	@ApiOperation("DB로 관리하기 애매한 enum 값들을 코드 기반으로 관리할 때")
	@GetMapping("enums")
	public List<PdSortMapperValue> getEnumList() {
		// Bean 사용 O
		return pdSortMapperBean.getPdSortList();
